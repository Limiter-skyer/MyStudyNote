# switch
```js
let weekday = "星期日"
switch (weekday) {
    case "星期日":
        document.writeln("今天是周日");
        break;
        
    default:
        document.writeln("你有问题");
        break;
}
```
# for循环
```js
for (let i = 0; i < 5; i++) {
    document.writeln("for循环语句")
}
```
# while循环
```js
let j = 0;
while (j < 5) {
    document.writeln("while循环语句");
    j++;
}
```
# do…while…循环
```js
let a = 0;
do {
    document.writeln("dowhile循环语句")
    a++;
} while (a < 5);
```
